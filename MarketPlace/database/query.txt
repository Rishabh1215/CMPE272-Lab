CREATE TABLE `shopingdb`.`user` ( `firstName` VARCHAR(50) NOT NULL,  `lastName` VARCHAR(50) NOT NULL , `emailId` VARCHAR(100) NOT NULL , `password` VARCHAR(50) NOT NULL,  `cellPhone` VARCHAR(50) NOT NULL,  `homePhone` VARCHAR(50) NOT NULL,  `address` VARCHAR(100) NOT NULL ) ENGINE = InnoDB;
CREATE TABLE `shopingdb`.`domain` ( `name` VARCHAR(100) NOT NULL , `visitCount` INT NOT NULL , `emailId` VARCHAR(100) NOT NULL , `productName` VARCHAR(100) NOT NULL, `imageUrl` VARCHAR(1000) NOT NULL, `origin` VARCHAR(1000) NOT NULL ) ENGINE = InnoDB;
CREATE TABLE `shopingdb`.`review` ( `review` VARCHAR(200) NOT NULL , `domain` VARCHAR(100) NOT NULL , `rating` INT NOT NULL , `productName` VARCHAR(100) NOT NULL , `emailId` VARCHAR(100) NOT NULL, `imageUrl` VARCHAR(1000) NOT NULL, `origin` VARCHAR(1000) NOT NULL) ENGINE = InnoDB;
CREATE TABLE `shopingdb`.`activities` ( `domain` VARCHAR(100) NOT NULL , `url` VARCHAR(200) NOT NULL , `emailId` VARCHAR(100) NOT NULL , `activity` VARCHAR(200) NOT NULL ) ENGINE = InnoDB;
